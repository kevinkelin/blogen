3. Strings are immutable
   
    When we say that a string is immutable, we mean that the string internally uses a pointer to point to a UTF-8 byte array, and the pointed byte array cannot be changed individually. doesn't mean the string points to a new UTF-8 array

    ```go
    package main

    import "fmt"

    func main() {
        var name string
        var name1 = []byte("yangyanxing Chinese characters")
        name = string(name1)
        fmt.Println(&name,name)

        name = "Yang Yanxing" //This is ok
        fmt.Println(&name) //name's address will not change

        //illegal
        //name[0] = "a"

    }

    ```

    If you want to change the value in the string, you need to first convert the string to a byte array (\[\]byte) or a character array (\[\]rune), and use a character array in the case of Chinese

    ```go
    package main

    import "fmt"

    func main() {
        var name = "yangyanxing"
        fmt.Println(&name,name)
        names := []byte(name)
        //Byte arrays need to use single quotes, double quotes are strings
        names[0] = 'f'
        fmt.Println(&name,string(names))

        name = "Yang Yanxing" //This is ok
        //If there are Chinese characters, you need to use a character array
        namer := []rune(name)
        // need to use single quotes
        namer[0] = 'rice'
        //name's address will not change
        fmt.Println(&name,string(namer))
    }

    ```

4. Convert between strings, byte arrays and character arrays

    ```go
    package main

    import "fmt"

    func main() {
        var name = "yangyanxing"
        fmt.Println(&name,name)
        names := []byte(name)
        fmt.Printf("%T\n",names)
        names[0] = 'f' //byte array needs to use single quotes, double quotes are strings
        name = string(names)
        fmt.Println(&name,name)

        name = "Yang Yanxing"
        //If there are Chinese characters, you need to use a character array
        namer := []rune(name)
        fmt.Printf("%T\n",namer)
        namer[0] = 'rice'
        name = string(namer)
        fmt.Println(&name,name) //The address of name will not change
    }

    ```

5. Traversal of strings

    Strings can be traversed by bytes or characters

    ```go
    package main

    import "fmt"

    func main() {
        var name = "yang Yang Yanxing"
        // loop through bytes
        for i:=0;i<len(name);i++{
            fmt.Printf("%v %T:%v\n",i,name[i],name[i])
        }

        fmt.Println("")

        // Traverse by character
        for _,r := range name{
            //%c is the Unicode character represented by the output value, without single quotes such as y
            //%q outputs the Unicode character (with single quotes) represented by the value, such as 'yang'
            fmt.Printf("%c",r)
        }
    }

    ```

    The result is

    ```

6. String concatenation

    You can use ` + ` to concatenate directly

    ```go
    package main

    import "fmt"

    func main() {
        // Strings can be concatenated with +
        var name = "yang" + "yan" + "xing"
        fmt.Println(name)
        //Use += to append a string to the original string
        name += "Yang Yanxing"
        fmt.Println(name)

        // + cannot be at the beginning of the line, only at the end of the line, just like the else in if else
        s := "BeiJing" +
            "TongZhou"
        fmt.Println(s)

    }

    ```

    Joining with the + sign in a loop is not efficient, you can use the `strings.Join` method

    ```go
    package main

    import (
        "fmt"
        "strings"
    )

    func main() {
        var names = []string{"yang","fan","zhang","li","chen"}
        n := strings. Join(names," ")
        fmt.Println(n)

        s := "yang&aaa&bbb&ccc"
        // split by &
        str := strings.Split(s,"&")
        fmt.Println(str)

    }

    ```

    >yang fan zhang li chen
    [yang aaa bbb ccc]

# Common methods of the strings package
This package provides common methods for string manipulation

```go
package main

import (
    "fmt"
    "strings"
)

func main() {
    s := "yang&yan&xing"
    //HasPrefix determines whether the string s starts with prefix
    fmt.Println(strings.HasPrefix(s,"y"))
    //HasSuffix determines whether the string s ends with suffix
    fmt.Println(strings.HasSuffix(s,"aaa"))
    //Contains determines whether the string s contains substr
    fmt.Println(strings.Contains(s,"yang"))
    //Replace replace the string, the last parameter is the number, -1 is to replace all
    fmt.Println(strings.Replace(s,"y","Y",-1))

    //ToLower converts all Unicode characters in the string to the corresponding lowercase characters
    ss := "AAA Yang Yanxing BbbbB"
    fmt.Println(strings.ToLower(ss))
    //ToUpper converts all Unicode characters in the string to the corresponding uppercase characters
    fmt.Println(strings.ToUpper(ss))

    //strings.TrimSpace(s) to strip whitespace at the beginning and end of the string
    //The middle one cannot be deleted
    sp := " aaa bbb ccc "
    fmt.Println(strings.TrimSpace(sp))
    //You can also remove the specified characters at the beginning or end
    ssp := "aaa Hey, how are you George?aaaa"
    fmt.Println(strings.TrimLeft(ssp,"He"))
    fmt.Println(strings.TrimRight(ssp,"ge"))
    fmt.Println(strings.Trim(ssp,"aaa"))

    //strings.Fields(s) will use
    //strings.Split(s, sep) is used to customize the split symbol to split the specified string
    sp2 := " aa a b bb ccc "
    fmt.Println(strings.Fields(sp2))
    fmt.Println(strings.Split(sp2," ")) //Use spaces to split

    //Join is used to concatenate slices whose element type is string using the split symbol to form a string
    sp3 := []string{"aa","a","b","bb","ccc"}
    fmt.Println(strings.Join(sp3,"$$$"))

}

package main

import (
"fmt"
"strconv"
)

func main() {
    //Itoa converts numbers to strings
    fmt.Println(strconv.Itoa(35))
    // Atoi converting strings to other types may not be successful
    fmt.Println(strconv.Atoi("100"))
    fmt.Println(strconv.ParseBool("true")) //Can succeed
    v,e := strconv.ParseBool("trues")//There will be err
    if e!=nil{
        fmt.Println(e.Error())
    }else{
        fmt.Println(v)
    }
}
