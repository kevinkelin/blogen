new and make in go language

Publish: 2019-08-13 Classify at: Go
Words:1519 Read:≈ 8min

Beginning to learn go language, I am a little confused about whether to use new or make when initializing variables, here is a brief summary

The difference between new() and make()

The following excerpt is from https://github.com/unknwon/the-way-to-go_ZH_CN/blob/master/eBook/07.2.md

Both allocate memory on the heap, but they behave differently for different types.

  • new(T) allocates a piece of memory for each new type T, initialized to 0 and returns a memory address of type *T: this method *returns a pointer to an address of type T with value 0 *, which applies to value types such as arrays and structs; it is equivalent to &T{}.
  • make(T) returns an initial value of type T, which only works with the 3 built-in reference types: slice, map and channel

In other words, the new function allocates memory and the make function initializes; the following diagram shows the difference:

From my understanding, new returns a pointer to a variable, but this pointer points to null, you cannot operate the pointer directly, otherwise an error will be reported unless you point the pointer to the address of a variable of this type. make returns a variable of this type, taking slice variables as an example. As mentioned above, make is suitable for creating slices, maps and channels, but new can also create

package main

import "fmt"

func main() {
	//Use make to create a slice, which returns the variable itself
	s1 := make([]int,5,10)
	fmt.Printf("use make create slise type %T value %v \n",s1,s1)
	s1[0] = 123
	s1[4] = 321
	fmt.Printf("make s1 type %T value %v \n",s1,s1)

	//# Use new to create a slice, which returns a pointer to the slice variable
	s2 := new([]int)
	fmt.Printf("use new create slise type %T value %v \n",s2,s2)
	fmt.Printf("new s2 type %T value %v \n",*s2,*s2)
	// If you want to assign a value, you need to use * dereference
	// Although no syntax error is reported here, if you try to use (*s2)[0] = 123 directly, there will be a runtime error,
	// panic: runtime error: index out of range
	s2 = &s1 //The variable pointer needs to point to the address of a variable of this type
	(*s2)[0] = 123
	(*s2)[4] = 3211
	fmt.Printf("new s2 type %T value %v \n",s2,s2)
	//The modification of s2 will also affect s1
	fmt.Printf("s1 type %T value %v \n",s1,s1)


	m1 := make(map[string]string)
	m1["name"] = "yangyanxing"
	m1["age"] = "30"
	fmt.Printf("m1 use make create type:%T value %v \n",m1,m1)

	m2 := new(map[string]string)
	fmt.Printf("m2 use new create type:%T value %v \n",m2,m2)
	//Direct assignment will report panic: assignment to entry in nil map
	m2 = &m1
	(*m2)["name"] = "fan"
	//Modifications to m2 will also affect m1
	fmt.Printf("after m2 change m1 value is %v",m1)

}

package main

import "fmt"

func main() {
	//Use make to initialize map
	mp2 := make(map[string]string)
	mp2["name"] = "yangyanxing"
	mp2["age"] = "18"
	fmt.Println("m2 address ",&mp2)
	// out:m2 address &map[name:yangyanxing age:18]

	mp3 := map[string]int{} //There must be {}, if there is no content in the curly brackets, it means that an empty dictionary is initialized
	mp3["yang"] = 18
	mp3["fan"] = 20
	fmt.Println(mp3)
	//out:map[yang:18 fan:20]

	mp4 := map[string]int{
	"yang":20,
	"fan":21, // even the last one must have a comma
	}
	fmt.Println(mp4)
	//out:map[yang:20 fan:21]

	mp5 := map[string]int{"yang":30} //Write on the same line without commas
	fmt.Println(mp5)
	//out: map[yang:30]

	mp6 := make(map[string]int,1) //You can also add a capacity to the map
	mp6["yang"] = 30
	fmt.Println("mp6 lens is ",len(mp6),"address:",&mp6)
	//out:mp6 lens is 1 address: &map[yang:30]
	mp6["fan"] = 31
	fmt.Println("mp6 lens is ",len(mp6),"address:",&mp6)
	//out:mp6 lens is 2 address: &map[yang:30 fan:31]

	//You can also use new, but you can't assign it directly, because it returns a null pointer at this time
	//You need to point to a variable address of this type before you can operate
	mp7 := new(map[string]int)
	fmt.Println(mp7)
	//out:&map[]

	//(*mp7)["yang"] = 100 //A runtime error will be reported
	mp7 = &mp6
	//mp7["fan"] = 1000 //You can't use mp7 directly, you need to use * to dereference first
	(*mp7)["yang"] = 100 //At this time, no runtime error will be reported

	fmt.Println(mp7)
	//out:&map[yang:100 fan:31]

}

package main

import "fmt"

func main() {
	//Use make to initialize the slice, you need to pass a len length, and the capacity cap is optional
	//If not passed, the length and capacity are the same
	sls1 := make([]int,5,10)
	sls1[0] = 100
	//append is appended to the end, which is a bit interesting here
	sls1 = append(sls1,200)
	// ... three dots, syntactic sugar in go, expand the previous slice
	sls1 = append(sls1,[]int{30,40}...)
	fmt.Println(sls1,len(sls1),cap(sls1))
	//out: [100 0 0 0 0 200 30 40] 8 10
	sls1 = append(sls1,3,4,6)//The expansion will occur after the original capacity of the slice is exceeded
	fmt.Println(sls1,len(sls1),cap(sls1))
	//out: [100 0 0 0 0 200 30 40 3 4 6] 11 20

	sls2 := make([]int,3)
	sls2[1] = 123
	fmt.Println(sls2,len(sls2),cap(sls2))
	//out: [0 123 0] 3 3

	// direct initialization
	sls3 := []int{}
	sls3 = append(sls3,10,20)
	fmt.Println(sls3,len(sls3),cap(sls3))
	//out: [10 20] 2 2

	sls4 := []int{1,2,3}
	sls5 :=[]int{
			1,
			2,
	5,//The comma here cannot be omitted
	}
	fmt.Println(sls4,sls5)
	//out: [1 2 3] [1 2 5]

	//Use new to create a slice, like map, it returns a pointer and cannot be directly manipulated
	// Need to point to the address of a variable first
	sls6 := new([]int)
	fmt.Println(sls6)//out: &[]
	sls6 = &sls4
	fmt.Println(sls6) //out: &[1 2 3]



}

package main

import "fmt"

type person struct {
	name string
	age int
}

func main() {

	fmt.Println("Array initialization")

	//declare and initialize an empty array whose element value is the zero value of the type
	arr1 := [2]int{}
	fmt.Println(arr1) //out: [0 0]

	//write the element value when initializing
	arr2 := [2]int{1,3}
	fmt.Println(arr2) //out: [1 3]
	//Only write one, not write zero value
	arr3 := [2]int{1}
	fmt.Println(arr3) //out: [1 0]

	//arr4 := make([2]int) //arrays cannot use make
	var arr5[2]int
	arr5[0] = 100
	fmt.Println(arr5)//[100 0]

	//Do not specify the size of the array, use ... three dots to automatically expand the calculation
	arr6 := [...]int{2,4,6,8}
	fmt.Println(arr6,len(arr6))//out: [2 4 6 8] 4

	//Create an array with new, and get a pointer
	arr7 := new([3]int)
	fmt.Println(arr7)//out: &[0 0 0]
	// can operate directly on the pointer
	arr7[0] = 3
	// same as dereferencing with *
	(*arr7)[1] = 4
	fmt.Println(arr7) //out: &[3 4 0]

}

package main

import "fmt"

type person struct {
	name string
	age int
}

func main() {

	fmt.Println("Initialization of structure")
	//Use new, return the structure pointer
	stru1 := new(person)
	fmt.Println(stru1)//out &{ 0} , the default is the zero value of the field
	//You can use this pointer directly to manipulate variables
	// same as dereferencing with *
	stru1.name = "yangyanxing"
	(*stru1).age = 18
	fmt.Println(stru1,*stru1) //&{yangyanxing 18} {yangyanxing 18}

	// Only one field is specified, no default zero value specified
	stru2 := person{name:"fan"}
	fmt.Println(stru2) //{fan 0}
	//All specified, the order of fields can be different from the order in which they were defined
	stru3 := person{age:18,name:"yang"}
	fmt.Println(stru3) //{yang 18}
	//Initialize in the order of the structure, at this time the value of the element must be written in full
	stru4 := person{"fan",17}
	fmt.Println(stru4) //{fan 17}

}
