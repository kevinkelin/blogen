from selenium import webdriver
from selenium.webdriver.common.action_chains import ActionChains #Introduce ActionChains mouse operation class
from selenium.webdriver.common.keys import Keys #Introduce keys class operation
import time

def s(int):
    time.sleep(int)
browser = webdriver.Chrome()
browser.get('http://www.baidu.com')
print 'Now maximize the browser'
browser.maximize_window()
text = browser.find_element_by_name('tj_duty').text
print text #Print filing information

browser.find_element_by_id('kw1').send_keys(u'Yang Yanxing')
print browser.find_element_by_id('kw1').get_attribute('type')
print browser.find_element_by_id('kw1').size #Print the size of the input box
browser.find_element_by_id('su1').click()
time.sleep(3)

print 'Now I will set the browser to be 480 wide and 800 high to display'
browser.set_window_size(480,800)
browser.get('http://m.mail.10086.cn')
time.sleep(3)

print 'Now I will go back to the previous page'
browser.maximize_window()
browser.back()
time.sleep(3)

print 'Now I will go back to the previous page'
browser.forward()
time.sleep(5)
print 'Now I will open Yang Yanxing's website for json search'
browser.get('http://static.yangyanxing.com')
browser.find_element_by_xpath(".//*[@id='ls']").send_keys(u'json')
browser.find_element_by_xpath(".//*[@id='header']/div[1]/div/form/input[2]").click()
time.sleep(5)
browser.quit()

browser = webdriver.Chrome()

print 'The following will be used to log in to Renren.com to perform the above comprehensive application'
browser.get('http://www.renren.com/SysHome.do')
browser.find_element_by_id('email').clear()#This is to select an element by id
browser.find_element_by_id('email').send_keys('email')
browser.find_element_by_id('email').send_keys(Keys.BACK_SPACE)
time.sleep(2)
browser.find_element_by_id('email').send_keys('m')
s(2)
browser.find_element_by_id('email').send_keys(Keys.CONTROL,'a')
s(2)
browser.find_element_by_id('email').send_keys(Keys.CONTROL,'x')#Cut off the content inside
s(2)
browser.find_element_by_id('email').send_keys(Keys.CONTROL,'v') #Re-enter it
s(2)
browser.find_element_by_name('password').clear()#This is to select the element by name
browser.find_element_by_name('password').send_keys('password')
#browser.find_element_by_xpath(".//*[@id='login']").click()#This is to select elements by xpath
browser.find_element_by_xpath(".//*[@id='login']").send_keys(Keys.ENTER) #Here you can log in by clicking the Enter key
browser.maximize_window()
article = browser.find_element_by_link_text(u'Zhou Bihua: What does it mean when there is a ghost in the Academy of Social Sciences?')
ActionChains(browser).move_to_element(article).perform()#Move the mouse here, but it is not easy to use here
ActionChains(browser).context_click(article).perform()
time.sleep(5)

browser.quit()

Begin to get in touch with selenium, which is widely used in web automation. The above are some of the most basic operations, starting a browser, opening a web page, forward and backward, positioning elements, keyboard input and mouse click operations, where xpath can apply firepath under firefox It can be obtained by plug-ins, but the limitations are relatively large, and there will be some compatibility problems. For example, the above code can be run under chrome, but some elements cannot be found under firefox, which is very depressing. . . keep exploring
